declare(strict_types=1);

namespace Infection\Logger\Html;

use Infection\Logger\Html\Node\Builder;
use Infection\Logger\Html\Node\Directory;
use Infection\Mutant\MetricsCalculator;
use Symfony\Component\Filesystem\Filesystem;

/**
 * @internal
 */
final class HtmlReport implements Report
{
    /**
     * @var ReportConfiguration
     */
    private $conf;

    /**
     * @var Filesystem
     */
    private $fs;

    /**
     * @var string
     */
    private $templatePath;

    public function __construct(ReportConfiguration $conf, Filesystem $fs)
    {
        $this->conf = $conf;
        $this->fs = $fs;
        $this->templatePath = __DIR__ . '/Template';
    }

    public function process(MetricsCalculator $metricsCalculator): void
    {
        $root = (new Builder())->build($metricsCalculator);
        $renderer = new Renderer($this->conf, $this->templatePath);

        if ($this->fs->exists($this->conf->getTargetDirectory())) {
            $this->fs->remove($this->conf->getTargetDirectory());
        }

        $this->dump($renderer, $root, $this->conf->getTargetDirectory());

        $this->fs->mirror($this->templatePath . '/css', $this->conf->getTargetDirectory() . '/.css');
        $this->fs->mirror($this->templatePath . '/img', $this->conf->getTargetDirectory() . '/.img');
        $this->fs->mirror($this->templatePath . '/js', $this->conf->getTargetDirectory() . '/.js');
    }

    private function dump(Renderer $renderer, Directory $root, string $target): void
    {
        $this->fs->mkdir($target);
        $this->fs->dumpFile($target . '/index.html', $renderer->render($root));

        foreach ($root->getChildren() as $child) {
            $this->dump($renderer, $child, $target . '/' . $child->getName());
        }

        foreach ($root->getFiles() as $child) {
            $this->fs->dumpFile($target . '/' . $child->getName() . '.html', $renderer->render($child));
        }
    }
}
