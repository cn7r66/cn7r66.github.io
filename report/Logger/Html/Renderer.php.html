declare(strict_types=1);

namespace Infection\Logger\Html;

use Infection\Logger\Html\Node\Directory;
use Infection\Logger\Html\Node\Node;
use Infection\Mutant\MetricsCalculator;
use Infection\Process\MutantProcess;
use Infection\Process\MutantProcessInterface;
use PackageVersions\Versions;

/**
 * @internal
 */
final class Renderer
{
    /**
     * @var ReportConfiguration
     */
    private $conf;

    /**
     * @var string
     */
    private $templatePath;

    public function __construct(ReportConfiguration $conf, string $templatePath)
    {
        $this->conf = $conf;
        $this->templatePath = $templatePath;
    }

    public function render(Node $node): string
    {
        $template = new \Text_Template($this->templatePath . '/index.html', '{{', '}}');
        $template->setVar([
            'path-to-root' => $node->pathToRoot(),
            'node-name' => $node->getName(),
            'breadcrumb' => $this->breadcrumb($node),
            'report-table' => $this->table($node),
            'page-content' => $this->pageContent($node),
            'footer' => $this->footer(),
        ]);

        return $template->render();
    }

    private function breadcrumb(Node $node): string
    {
        $items = [$this->breadcrumbItem($node->getName(), true)];

        $pathToParent = $node->pathToParent();

        while ($node->hasParent()) {
            $node = $node->getParent();
            $link = $this->link($pathToParent, $node->getName());
            $items[] = $this->breadcrumbItem($link);
            $pathToParent = $node->pathToParent() . $pathToParent;
        }

        $breadcrumb = new \Text_Template($this->templatePath . '/breadcrumb.html', '{{', '}}');
        $breadcrumb->setVar([
            'breadcrumb-items' => implode("\n", array_reverse($items)),
        ]);

        return $breadcrumb->render();
    }

    private function breadcrumbItem(string $nodeName, bool $active = false): string
    {
        $class = $active ? 'active' : '';

        $item = new \Text_Template($this->templatePath . '/breadcrumb-item.html', '{{', '}}');
        $item->setVar([
            'node-name' => $nodeName,
            'item-class' => $class,
        ]);

        return $item->render();
    }

    private function table(Node $node): string
    {
        $table = new \Text_Template($this->templatePath . '/table.html', '{{', '}}');
        $table->setVar([
            'table-items' => $this->tableItems($node),
        ]);

        return $table->render();
    }

    private function tableItems(Node $node): string
    {
        $items = [$this->tableItem($node->getName(), $node->getMetricsCalculator())];

        if ($node instanceof Directory) {
            foreach ($node as $child) {
                $link = $this->iconLink($child);
                $items[] = $this->tableItem($link, $child->getMetricsCalculator());
            }
        }

        return implode("\n", $items);
    }

    private function tableItem(string $nodeName, MetricsCalculator $metrics): string
    {
        $item = new \Text_Template($this->templatePath . '/table-item.html', '{{', '}}');
        $item->setVar([
            'node-name' => $nodeName,
            'mutation-score-color' => $this->scoreColor($metrics->getMutationScoreIndicator()),
            'mutation-coverage-color' => $this->scoreColor($metrics->getCoverageRate()),
            'covered-code-color' => $this->scoreColor($metrics->getCoveredCodeMutationScoreIndicator()),
            'mutation-score' => $metrics->getMutationScoreIndicator(),
            'mutation-coverage' => $metrics->getCoverageRate(),
            'covered-code' => $metrics->getCoveredCodeMutationScoreIndicator(),
            'killed' => $metrics->getKilledCount(),
            'escaped' => $metrics->getEscapedCount(),
            'timeout' => $metrics->getTimedOutCount(),
            'no-coverage' => $metrics->getNotCoveredByTestsCount(),
            'runtime-error' => $metrics->getErrorCount(),
            'total-detected' => $metrics->getTotalMutantsCount() - ($metrics->getEscapedCount() +
                                                                    $metrics->getNotCoveredByTestsCount()),
            'total-undetected' => $metrics->getEscapedCount() + $metrics->getNotCoveredByTestsCount(),
            'total-mutants' => $metrics->getTotalMutantsCount(),
        ]);

        return $item->render();
    }

    private function scoreColor(float $score): string
    {
        if ($score < $this->conf->getMinMsi()) {
            return 'danger';
        }

        if ($score >= $this->conf->getMinMsi() &&
            $score < $this->conf->getMaxMsi()) {
            return 'warning';
        }

        return 'success';
    }

    private function link(string $url, string $text): string
    {
        $link = new \Text_Template($this->templatePath . \DIRECTORY_SEPARATOR . 'link.html', '{{', '}}');
        $link->setVar([
            'link-url' => $url,
            'link-text' => $text,
        ]);

        return $link->render();
    }

    private function iconLink(Node $node): string
    {
        $icon = new \Text_Template($this->templatePath . \DIRECTORY_SEPARATOR . 'icon.html', '{{', '}}');
        $icon->setVar([
            'path-to-root' => $node->getParent()->pathToRoot(),
            'icon-name' => $node instanceof Directory ? 'directory' : 'file',
        ]);

        return $this->link($node->url(), $icon->render() . $node->getName());
    }

    private function pageContent(Node $node): string
    {
        if ($node instanceof Directory) {
            return '';
        }

        $source = new \Text_Template($this->templatePath . '/source.html', '{{', '}}');
        $source->setVar([
            'source-code' => htmlentities(file_get_contents($node->getMetricsCalculator()->getAllMutantProcesses()[0]->getOriginalFilePath())),
            'mutant-table' => $this->mutantTable($node),
        ]);

        return $source->render();
    }

    private function mutantTable(Node $node): string
    {
        $table = new \Text_Template($this->templatePath . '/mutant-table.html', '{{', '}}');
        $table->setVar([
            'mutant-table-items' => $this->mutantTableItems($node->getMetricsCalculator()->getAllMutantProcesses())
        ]);

        return $table->render();
    }

    /**
     * @param MutantProcessInterface[]
     */
    private function mutantTableItems(array $processes): string
    {
        $items = [];
        foreach ($processes as $number => $process) {
            $items[] = $this->mutantTableItem($number, $process);
        }

        return implode("\n", $items);
    }

    private function mutantTableItem(int $number, MutantProcessInterface $process): string
    {
        $item = new \Text_Template($this->templatePath . '/mutant-table-item.html', '{{', '}}');
        $item->setVar([
            'number' => $number,
            'mutator' => $process->getMutator()::getName(),
            'state' => $this->mutantState($process->getResultCode()),
            'location' => $process->getOriginalStartingLine(),
        ]);

        return $item->render();
    }

    private function mutantState(int $status): string
    {
        switch ($status) {
            case MutantProcess::CODE_KILLED:
                return 'Killed';
            case MutantProcess::CODE_ESCAPED:
                return 'Escaped';
            case MutantProcess::CODE_ERROR:
                return 'Error';
            case MutantProcess::CODE_TIMED_OUT:
                return 'Timeout';
            case MutantProcess::CODE_NOT_COVERED:
                return 'Not Covered';
            default:
                return 'Unknown';
        }
    }

    private function footer(): string
    {
        $matches = [];
        preg_match('/(.*)@/', Versions::getVersion('infection/infection'), $matches);

        $footer = new \Text_Template($this->templatePath . '/footer.html', '{{', '}}');
        $footer->setVar([
            'lower-bound' => $this->conf->getMinMsi(),
            'upper-bound' => $this->conf->getMaxMsi(),
            'version' => $matches[1],
        ]);

        return $footer->render();
    }
}
